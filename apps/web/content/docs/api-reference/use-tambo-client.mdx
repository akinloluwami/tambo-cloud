<Callout type="info" title="Quick Example: Advanced Client Access">
  Need to drop down and call a low-level method? Grab the raw client with
  `useTamboClient()` and invoke anything exposed by the SDK.

```tsx title="advanced-client.tsx"
"use client";

import { useTamboClient } from "@tambo-ai/react";

export function AdvancedExample() {
  const { client } = useTamboClient();

  async function handleClick() {
    // Direct SDK call â€“ not wrapped by a convenience hook.
    await client.someAdvancedMethod({ force: true });
  }

  return (
    <button onClick={handleClick} className="btn-primary">
      Run advanced action
    </button>
  );
}
```

</Callout>
